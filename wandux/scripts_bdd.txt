CREATE TABLE `global_conf` (
  `key` varchar(50) NOT NULL default '',
  `hostname` varchar(50) NOT NULL default '',
  PRIMARY KEY  (`key`)
) TYPE=MyISAM;


CREATE TABLE `network_dhcp_config` (
  `key` varchar(50) NOT NULL default '',
  `dhcp_server` varchar(16) NOT NULL default '',
  `dhcp_def_gateway` varchar(16) NOT NULL default '',
  `dhcp_domain` varchar(50) NOT NULL default '',
  `dhcp_adress` varchar(16) NOT NULL default '',
  `dhcp_pri_dns` varchar(16) NOT NULL default '',
  `dhcp_sec_dns` varchar(16) NOT NULL default '',
  `dhcp_subnetmask` varchar(16) NOT NULL default '',
  PRIMARY KEY  (`key`)
) TYPE=MyISAM;



CREATE TABLE `users_data` (
  `key` varchar(50) NOT NULL default '',
  `login` varchar(50) NOT NULL default '',
  `type` varchar(50) NOT NULL default '',
  PRIMARY KEY  (`key`)
) TYPE=MyISAM;
